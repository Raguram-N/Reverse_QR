<script>
// Configuration mapping core deterministic static parameters arrays mock
const pinSequence = ["4021", "8934", "1105", "5762", "9319"];
let currentSequenceIndex = 0;
let inputBuffer = "";
let systemState = "SCAN_REQUIRED"; // SCAN_REQUIRED, PIN_PROMPT, AUTH_SUCCESS, FAILURE

function initPrototype() {
renderPinPool();
updateTerminal();
}

function renderPinPool() {
const poolContainer = document.getElementById("pinPool");
poolContainer.innerHTML = "";

pinSequence.forEach((pin, index) => {
const pill = document.createElement("div");
pill.classList.add("pin-pill");
pill.innerText = `PIN #${index + 1}: ${pin}`;

if (index < currentSequenceIndex) {
pill.classList.add("burned");
pill.innerText += " [BURNED]";
} else if (index === currentSequenceIndex) {
pill.classList.add("active");
pill.innerText += " [ACTIVE]";
}
poolContainer.appendChild(pill);
});
}

function updateTerminal() {
const statusEl = document.getElementById("terminalStatus");
const displayEl = document.getElementById("terminalDisplay");

if (systemState === "SCAN_REQUIRED") {
statusEl.innerText = "AUTHENTICATION CHANNEL CLOSED";
statusEl.style.color = "#fb923c";
displayEl.innerText = "👉 Click Card QR to Scan";
displayEl.style.cursor = "pointer";
// Allow clicking the graphic block card element placeholder to trigger scan event action sequence flow
document.querySelector('.token-card').onclick = triggerScanAction;
} else if (systemState === "PIN_PROMPT") {
statusEl.innerText = "TOKEN VALIDATED: ENTER ACTIVE PIN";
statusEl.style.color = "#4ade80";
displayEl.innerText = "".padStart(inputBuffer.length, "*") || "ENTER 4-DIGIT PIN";
document.querySelector('.token-card').onclick = null;
} else if (systemState === "AUTH_SUCCESS") {
statusEl.innerText = "TX COMPLETED STATUS: OK";
statusEl.style.color = "#4ade80";
displayEl.innerText = "✅ Settle Success! [Rs. 250]";
} else if (systemState === "FAILURE") {
statusEl.innerText = "HARDWARE VALIDATION CRITICAL EXCEPTION";
statusEl.style.color = "#f87171";
displayEl.innerText = "❌ Security Replay Alert";
}
}

function triggerScanAction() {
if (systemState !== "SCAN_REQUIRED") return;

if (currentSequenceIndex >= pinSequence.length) {
logMessage("CRITICAL EXCEPTION: All sequential hashes exhausted. Please refresh token index bounds via app authorization.");
systemState = "FAILURE";
updateTerminal();
return;
}

logMessage("Merchant hardware peripheral triggered scan framework on static target.");
logMessage(`Routing transaction verification path for internal pointer token index ref: [Slot #${currentSequenceIndex + 1}]`);
systemState = "PIN_PROMPT";
updateTerminal();
}

function pressKey(val) {
if (systemState !== "PIN_PROMPT") {
logMessage("System operation trace exception: Scan initialization vector step required before input context generation.");
return;
}

if (val === "C") {
inputBuffer = "";
updateTerminal();
} else if (val === "E") {
validateInputPIN();
} else {
if (inputBuffer.length < 4) {
inputBuffer += val;
updateTerminal();
}
}
}

function validateInputPIN() {
if (inputBuffer.length !== 4) {
logMessage("Validation failure: Length specification profile out of baseline limits.");
return;
}

const expectedPIN = pinSequence[currentSequenceIndex];
logMessage(`Transmitting current entry payload to centralized routing switch mapper...`);

if (inputBuffer === expectedPIN) {
logMessage(`MAPPING MATCH FOUND: Hash verified successfully.`);
logMessage(`BURN CORE MECHANISM EXECUTED: Invalidating entry parameter index string [${expectedPIN}] permanently.`);

currentSequenceIndex++;
systemState = "AUTH_SUCCESS";
renderPinPool();
updateTerminal();

setTimeout(() => {
if (currentSequenceIndex < pinSequence.length) {
systemState = "SCAN_REQUIRED";
inputBuffer = "";
logMessage("Pipeline interface cycled safely to next active sequential entry tracking array slot.");
updateTerminal();
} else {
logMessage("NOTICE: Token lifecycle storage index bound exhaustion achieved.");
systemState = "SCAN_REQUIRED";
updateTerminal();
}
}, 3000);

} else {
logMessage(`SECURITY CRITICAL MATCH ERROR: Received invalid input structure string credential. Target sequence value preserved intact.`);
inputBuffer = "";
systemState = "SCAN_REQUIRED";
updateTerminal();
}
}

function logMessage(msg) {
const logBox = document.getElementById("logBox");
const timestamp = new Date().toLocaleTimeString();
logBox.innerHTML += `<div>[${timestamp}] ${msg}</div>`;
logBox.scrollTop = logBox.scrollHeight;
}

function resetSimulation() {
currentSequenceIndex = 0;
inputBuffer = "";
systemState = "SCAN_REQUIRED";
document.getElementById("logBox").innerHTML = "System tracking states storage variables cleared safely. Simulation restarted.";
initPrototype();
}

// Trigger lifecycle mapping on instantiation phase entry array mapping
window.onload = initPrototype;
